cmake_minimum_required(VERSION 3.1.0 FATAL_ERROR)
project( TipTilt )
set(CMAKE_BUILD_TYPE Debug)
find_package( OpenCV REQUIRED )
set (INCLUDE_DIR ${CMAKE_SOURCE_DIR}/include)
include_directories( ${OpenCV_INCLUDE_DIRS} ${INCLUDE_DIR})
set(SOURCES src/main.cpp src/CameraAndImageHandling.cpp src/TipTilt.cpp ${INCLUDE_DIR}/CameraAndImageHandling.hpp ${INCLUDE_DIR}/TipTilt.hpp)
add_executable( TipTilt ${SOURCES} )
target_compile_features( TipTilt PRIVATE cxx_thread_local)
target_link_libraries( TipTilt ${OpenCV_LIBS} -lpthread)
